from lshash import LSHash

vocablist = ['a','b','c','d','e','f','g','h','i','j','k','l','m','n','o','p','q','r','s','t','u','x','y','z']
dataSet = ['desk','function','see','sky','sea']

lsh = LSHash(10, len(vocablist))

def bagof_word2vec(vocablist, inputset):
    returnvec = [0] * len(vocablist)
    for word in inputset:
        if word in vocablist:
            returnvec[vocablist.index(word)] += 1
        else:
            print('word:', word, 'is not in the list_vec')
    return returnvec

for postinDoc in dataSet:
    trainMat_vec = bagof_word2vec(vocablist,postinDoc)
    lsh.index(trainMat_vec)

test_str = 'functior'
test_vec = bagof_word2vec(vocablist,test_str)
